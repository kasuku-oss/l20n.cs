28  public static bool PeekAndParse(CharStream cs, out FTL.AST.Pattern pattern)
29  {
30  // A quoted-pattern is the easiest to detect, so we'll try that first
31  if(cs.PeekNext() == '"') {
32  pattern = ParseQuoted(cs);
33  return true;
34  }
35 
36  // it might still be an unquoted-pattern, but this is trickier to detect
37  // first let's try to detect a placeable and block-text
38  if(Placeable.Peek(cs) || AnyText.PeekBlockText(cs)) {
39  pattern = ParseUnquoted(cs);
40  return true;
41  }
42 
43  // if not any of the above, the only thing left is unquoted-text
44  int bufferPos = cs.Position;
45  WhiteSpace.Parse(cs);
46  char next = cs.PeekNext();
47  cs.Rewind(bufferPos);
48  if(!CharStream.IsEOF(next) && !CharStream.IsNL(next)) {
49  pattern = ParseUnquoted(cs);
50  return true;
51  }
52 
53  pattern = null;
54  return false;
55  }

57  // '"' (placeable | quoted-text)* '"'
58  public static FTL.AST.Pattern ParseQuoted(CharStream cs)
59  {
60  cs.SkipCharacter('"');
61  FTL.AST.Pattern pattern = new FTL.AST.Pattern(true);
62  FTL.AST.INode child;
63 
64  while(cs.PeekNext() != '"') {
65  if(Placeable.PeekAndParse(cs, out child)) {
66  pattern.AddChild(child);
67  continue;
68  }
69 
70  // it's not a placeable, and we haven't seen a quote,
71  // so it must be a quoted-text
72  child = AnyText.ParseQuoted(cs);
73  pattern.AddChild(child);
74  }
75 
76  cs.SkipCharacter('"');
77  return pattern;
78  }

80  // (unquoted-text | placeable | block-text)+
81  public static FTL.AST.Pattern ParseUnquoted(CharStream cs)
82  {
83  FTL.AST.Pattern pattern = new FTL.AST.Pattern(false);
84  FTL.AST.INode child = ParseUnquotedChild(cs);
85  if(child == null) {
86  throw cs.CreateException(
87  "no unquoted child could be found, while at least one was expected", null);
88  }
89 
90  do {

92  child = ParseUnquotedChild(cs);
93  } while(child != null);

95  return pattern;
96  }

98  private static FTL.AST.INode ParseUnquotedChild(CharStream cs)
99  {
100  FTL.AST.INode child;
101 
102  if(Placeable.PeekAndParse(cs, out child))
103  return child;
104 
105  if(AnyText.PeekAndParseBlock(cs, out child))
106  return child;
107 
108  // as long as we don't have a newline char,
109  // we'll assume it's an unquoted-child
110  if(AnyText.PeekUnquoted(cs))
111  return AnyText.ParseUnquoted(cs);
112 
113  // return null if no child could be found
114  return null;
115  }
