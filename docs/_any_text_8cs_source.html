73  public static bool PeekBlockText(CharStream cs)
74  {
75  char next = cs.PeekNext();
76  // if next char isn't a NewLine Character, we know for sure we
77  // are not dealing with a block-text
78  if(CharStream.IsEOF(next) || !CharStream.IsNL(next))
79  return false;
80 
81  // from here on out, we're still not sure if we're dealing with a block-text
82  // thus we start buffering so we can go back in time
83  // here we check if we have the following pattern: `NL __ '|'`
84  int bufferPos = cs.Position;
85  NewLine.Parse(cs);
86  WhiteSpace.Parse(cs);
87 
88  // if the next unbuffered char is not '|' we're not dealing with a block-text
89  // and can return;
90  next = cs.PeekNext();
91  cs.Rewind(bufferPos);
92  return next == '|';

97  {
98  char next = cs.PeekNext();

100  // are not dealing with a block-text
101  if(CharStream.IsEOF(next) || !CharStream.IsNL(next)) {
102  result = null;
103  return false;
104  }
105 
106  // from here on out, we're still not sure if we're dealing with a block-text
107  // thus we start buffering so we can go back in time
108  // here we check if we have the following pattern: `NL __ '|'`
109  int bufferPos = cs.Position;
110  NewLine.Parse(cs);
111  WhiteSpace.Parse(cs);
112 
113  // if the next unbuffered char is not '|' we're not dealing with a block-text
114  // and can return;
115  if(cs.PeekNext() != '|') {
116  cs.Rewind(bufferPos);
117  result = null;
118  return false;
119  }
120 
121  // we know for sure we're dealing with a block-text,
122  // buffer can be flushed and we can start checking for more lines as well;
123  FTL.AST.INode line;
124 
125  FTL.AST.BlockText blockText = new FTL.AST.BlockText();
126  do {
127  cs.SkipNext(); // skip '|'
128

130 
131  if(!Placeable.PeekAndParse(cs, out line)) {

133  line = ParseUnquoted(cs);
134  }
135

137  blockText.AddLine(line);
138

140  // otherwise we can stop early with trying
141  next = cs.PeekNext();
142  if(CharStream.IsEOF(next) || !CharStream.IsNL(next))
143  break;
144 
145  // check if we have more lines
146  bufferPos = cs.Position;
147  NewLine.Parse(cs);
148  WhiteSpace.Parse(cs);
149
