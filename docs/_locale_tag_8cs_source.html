32  // set Language

35  } else {
36  throw new InputException(
37  "`{0}` is an invalid language, contains non-alphabetic characters", parts[0]);
38  }

66  public int Compare(LocaleTag requested)
67  {
68  // if the language doesn't match, than a match is not possible
69  if(m_Language != requested.m_Language)
70  return 0;
71 
72  // if the region doesn't match,
73  // we give a higher value if this localeTag is a pure-language tag.
74  if(m_Region != requested.m_Region) {
75  if(m_Region == null)
76  return 2;
77  return 1;
78  }
79 
80  // if the extension doesn't match,
81  // we give a higher value if this localeTag is a pure-languageRegion tag.
82  if(m_Extension != requested.m_Extension) {
83  if(m_Extension == null)
84  return 4;
85  return 3;
86  }

88  // a perfect match,
